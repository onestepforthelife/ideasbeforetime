function showApprovalForm() {
    document.getElementById('approvalForm').style.display = 'block';
    document.getElementById('paymentOptions').style.display = 'none';
}

function hideApprovalForm() {
    document.getElementById('approvalForm').style.display = 'none';
    document.getElementById('paymentOptions').style.display = 'block';
}

function submitApprovalRequest() {
    const name = document.getElementById('approvalName').value;
    const email = document.getElementById('approvalEmail').value;
    const company = document.getElementById('approvalCompany').value;
    const role = document.getElementById('approvalRole').value;
    const purpose = document.getElementById('approvalPurpose').value;
    
    if (!name || !email || !role || !purpose) {
        alert('Please fill all required fields');
        return;
    }
    
    // Get user's profile data from previous steps
    const profileData = {
        platform: document.getElementById('platform').value,
        profileLink: document.getElementById('profileLink').value,

        industry: document.getElementById('industry').value
    };
    
    // Create email body
    const emailBody = `
APPROVAL REQUEST - Social Profile Optimizer

${purpose}
