🚀 STEP 1: LOG INTO CLOUDFLARE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Go to: https://dash.cloudflare.com
2. Log in with your Cloudflare account
3. You'll see your sites listed

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
🚀 STEP 2: SELECT YOUR SITE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Click on: ideasbeforetime.pages.dev

(Or if you have a custom domain, click that)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
🚀 STEP 3: GO TO SECURITY SETTINGS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

In the left sidebar, click: Security

You'll see several security options.

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
🚀 STEP 4: ENABLE WAF (WEB APPLICATION FIREWALL)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Click: WAF (Web Application Firewall)

2. You'll see "Managed Rules" section

3. Toggle ON: "Cloudflare Managed Ruleset"
   - This enables Cloudflare's automatic protection
   - Blocks common attacks (XSS, SQL injection, etc.)
   - FREE on all plans

4. Toggle ON: "Cloudflare OWASP Core Ruleset"
   - Industry-standard security rules
   - Protects against OWASP Top 10 vulnerabilities
   - FREE on all plans

5. Click "Save" or "Deploy"

✅ WAF is now active!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
🚀 STEP 5: SET SECURITY LEVEL TO HIGH
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Still in Security section, scroll down

2. Find: "Security Level"

3. Change from "Medium" to "High"
   - This makes Cloudflare more aggressive in blocking threats
   - May challenge suspicious visitors with CAPTCHA
   - Recommended for maximum protection

4. Click "Save"

✅ Security Level is now High!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
🚀 STEP 6: VERIFY SSL/TLS (Should Already Be Active)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. In left sidebar, click: SSL/TLS

2. Verify you see: "Active Certificate" with a green checkmark ✅

3. Encryption mode should be: "Full" or "Full (strict)"

If not already set:
- Change to: "Full (strict)"
- Click "Save"

✅ SSL/TLS verified!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
🚀 STEP 7: ENABLE BOT FIGHT MODE (Optional but Recommended)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Go back to: Security

2. Scroll down to: "Bots"

3. Toggle ON: "Bot Fight Mode"
   - Blocks malicious bots automatically
   - FREE on all plans
   - Doesn't affect legitimate users

4. Click "Save"

✅ Bot protection active!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
✅ STEP 8: VERIFY EVERYTHING IS ACTIVE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
