    let scrollCount = 0;
    let lastHeight = document.body.scrollHeight;
    let noChangeCount = 0;
    
    const scrollInterval = setInterval(() => {
        // Scroll to bottom
        window.scrollTo(0, document.body.scrollHeight);
        scrollCount++;
